submitBtn.disabled = false; } }); // Загрузка теста из БД async function loadTest(testId) { try { const response = await fetch(`api.php?action=get_test&test_id=${testId}`); const result = await response.json(); } } // Обновленная функция calculateResult с сохранением window.calculateResult = function() { let correct = 0; window.testData.questions.forEach((q, index) => { if (window.userAnswers[index] === q.correct) { correct++;
